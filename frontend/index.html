SELECT   t1.customer_name 
FROM     customers   AS t1 
JOIN     orders      AS t2 
JOIN     order_items AS t3 
JOIN     products    AS t4 
WHERE    t4.product_name = “food” 
GROUP BY t1.customer_id 
HAVING   COUNT(*) >= 1
